function foo(x, y, z){
    let a = x + 1;
    let b = a + y;
    let c = 0;
    
    while (a < z) {
        c = a + b;
        z = c * 2;
    }
    
    return z;
}

function foo(x, y, z){

    let b = a + y;
    let c = 0;
